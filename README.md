# npm初始化

    npm init

# 安装webpack和babel

    npm install --save-dev webpack babel-core babel-loader babel-preset-react babel-preset-es2015

# 游戏发布

    npm run build

# 启动本地服务器，9999是端口号，可任意修改

    python -m SimpleHTTPServer 9999

# 游戏测试
打开浏览器，输入以下URL，9999是上一步中设定的端口号

    http://localhost:9999/build/
